    // 1. 상태 및 임시 데이터 변수 정의
    let tasks = [];            // 할일들을 담아둘 바구니
    let currentFilter = 'all'; // 기본 필터 설정 (전체보기)
    let currentDate = new Date(); // 현재 보고 있는 달력의 기준 날짜
    
    // 프로필 기본 상자값
    let profileData = { name: '', quote: '', photo: '' };

    // 2. 기초 유틸리티 기능들 (토스트 알림, 화면 전환)
    function showToast(message, duration = 2000) {
      const t = document.getElementById('toast');
      t.textContent = message;
      t.classList.add('show');
      setTimeout(() => t.classList.remove('show'), duration);
    }

    function showPage(pageId, navBtn, title) {
      document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
      document.querySelectorAll('.nav-item').forEach(b => b.classList.remove('active'));
      document.getElementById(pageId).classList.add('active');
      if (navBtn) navBtn.classList.add('active');
      document.getElementById('sidebar').classList.remove('open');
      document.getElementById('mobilePageTitle').textContent = title || '나의 앱';
      
      // 페이지가 바뀔 때 최신 데이터 상태로 다시 그려주기
      if (pageId === 'taskPage') renderTaskList();
      if (pageId === 'calendarPage') renderCalendar();

    }

    }

    function openModal(title, message, onConfirm) {
      document.getElementById('modalTitle').textContent = title;
      document.getElementById('modalMessage').textContent = message;

      document.getElementById('modalConfirmBtn').onclick = () => { onConfirm(); closeModal(); };

    function closeModal() {
      document.getElementById('modal').classList.remove('open');
    }

    function showSettingsSection(key, btn) {
      document.querySelectorAll('.settings-section').forEach(s => s.classList.remove('active'));
      document.querySelectorAll('.settings-tab').forEach(b => b.classList.remove('active'));

    // [추가] 할일 등록하기
    function addTask() {
      const input = document.getElementById('todoInput');
      const dateInput = document.getElementById('todoDateInput');
      
      const text = input.value.trim();
      const date = dateInput.value; // 연-월-일 형식으로 가져옴

      if (!text) {
        showToast('❌ 할 일을 입력해 주세요!');
        return;
      }
      if (!date) {
        showToast('❌ 목표 날짜를 골라주세요!');
        return;
      }

      // 새 할일 카드 묶음 만들기
      const newTask = {
        id: Date.now(), // 컴퓨터가 식별할 고유 시각 숫자
        text: text,
        date: date,
        status: 'todo' // 기본 설정은 시작전('todo') 상태
      };

      tasks.push(newTask);
      saveTasks();        // 창고에 저장
      renderTaskList();   // 목록 다시 그리기
      
      input.value = '';   // 입력창 청소
      showToast('✅ 할 일이 추가되었습니다!');
    }

    // [상태 전환] 시작전 -> 진행중 -> 완료 상태를 빙글빙글 토글하는 기능
    function toggleTaskStatus(id) {
      const task = tasks.find(t => t.id === id);
      if (!task) return;

      if (task.status === 'todo') {
        task.status = 'doing';
        showToast('⚡ 진행 중으로 변경했어요!');
      } else if (task.status === 'doing') {
        task.status = 'done';
        showToast('🎉 축하합니다! 완료 상태로 변경 완료!');
      } else {
        task.status = 'todo';
        showToast('↩️ 다시 시작 전으로 되돌렸어요.');
      }

      saveTasks();
      renderTaskList();
    }

    // [삭제] 할일 버리기
    function deleteTask(id) {
      tasks = tasks.filter(t => t.id !== id);
      saveTasks();
      renderTaskList();
      showToast('🗑 할 일이 삭제되었습니다.');
    }

    // [필터] 전체/시작전/진행중/완료 보기 전환
    function changeFilter(filterType) {
      currentFilter = filterType;
      document.querySelectorAll('.filter-tab').forEach(tab => tab.classList.remove('active'));
      document.getElementById('filter-' + filterType).classList.add('active');
      renderTaskList();
    }

    // [그리기] 사용자가 추가한 할일 목록 화면 채워주기
    function renderTaskList() {
      const container = document.getElementById('taskListContainer');
      container.innerHTML = ''; // 기존 리스트 리셋

      // 필터 타입에 맞춰서 걸러내기
      let filteredTasks = tasks;
      if (currentFilter !== 'all') {
        filteredTasks = tasks.filter(t => t.status === currentFilter);
      }

      // 등록된 데이터가 아무것도 없을 때 안내문구
      if (filteredTasks.length === 0) {
        container.innerHTML = `<p style="color: var(--text-tertiary); text-align: center; padding: 40px 0;">해당되는 할 일이 없습니다.</p>`;
        return;
      }

      // 데이터 정렬: 날짜 순서대로 정렬해서 보기 편하게 만듦
      filteredTasks.sort((a,b) => new Date(a.date) - new Date(b.date));

      // 각 데이터 카드를 화면에 찍어주기
      filteredTasks.forEach(task => {
        const item = document.createElement('div');
        item.className = 'task-item' + (task.status === 'done' ? ' completed' : '');
        
        // 상태별 예쁜 버튼 글씨 세팅
        let statusText = '시작 전';
        if (task.status === 'doing') statusText = '진행 중';
        if (task.status === 'done') statusText = '완료';

        item.innerHTML = `
          <div class="task-item-left">
            <span class="task-text">${task.text}</span>
            <span class="task-date">📅 ${task.date}</span>
          </div>
          <div style="display:flex; gap: 8px; align-items:center;">
            <button class="status-btn ${task.status}" onclick="toggleTaskStatus(${task.id})">${statusText}</button>
            <button class="btn-danger" style="padding: 4px 8px; font-size:11px;" onclick="deleteTask(${task.id})">삭제</button>
          </div>
        `;
        container.appendChild(item);
      });
    }

    // CALENDAR FUNCTION: 달력 자동 생성 및 연동 구현
    // =====================================================
    function moveMonth(direction) {
      currentDate.setMonth(currentDate.getMonth() + direction);
      renderCalendar();
    }

    function renderCalendar() {
      const grid = document.getElementById('calendarGrid');
      const title = document.getElementById('calendarMonthTitle');
      grid.innerHTML = ''; // 옛날 달력 청소

      const year = currentDate.getFullYear();
      const month = currentDate.getMonth();

      title.textContent = `${year}년 ${month + 1}월`;

      // 달력 헤더 (일~토) 깔기
      const weekDays = ['일', '월', '화', '수', '목', '금', '토'];
      weekDays.forEach(d => {
        const hCell = document.createElement('div');
        hCell.className = 'calendar-header-cell';
        hCell.textContent = d;
        grid.appendChild(hCell);
      });

      // 이번 달의 첫날과 마지막 날짜 구하기
      const firstDayIdx = new Date(year, month, 1).getDay();
      const lastDate = new Date(year, month + 1, 0).getDate();
      
      // 지난달 마지막 날짜 구하기 (빈칸 메꾸기용)
      const prevLastDate = new Date(year, month, 0).getDate();

      // 1. 지난달 날짜 채워넣기 (비활성화 스타일)
      for (let i = firstDayIdx - 1; i >= 0; i--) {
        const cell = document.createElement('div');
        cell.className = 'calendar-day-cell other-month';
        cell.innerHTML = `<span class="day-number">${prevLastDate - i}</span>`;
        grid.appendChild(cell);
      }

      // 2. 이번 달 날짜 채워넣기
      const today = new Date();
      for (let date = 1; date <= lastDate; date++) {
        const cell = document.createElement('div');
        cell.className = 'calendar-day-cell';
        
        // 오늘 날짜 하이라이트 효과
        if (today.getFullYear() === year && today.getMonth() === month && today.getDate() === date) {
          cell.classList.add('today');
        }

        cell.innerHTML = `<span class="day-number">${date}</span>`;

        // 이 날짜에 해당하는 마감 할 일이 있는지 서치하기
        // 날짜를 컴퓨터가 읽을 수 있는 '2026-06-22' 형태로 포맷팅
        const currentCellStr = `${year}-${String(month + 1).padStart(2, '0')}-${String(date).padStart(2, '0')}`;
        const dayTasks = tasks.filter(t => t.date === currentCellStr);

        // 발견된 할 일이 있다면 달력 칸 안에 작은 뱃지로 심어두기
        dayTasks.forEach(t => {
          const todoBadge = document.createElement('div');
          todoBadge.className = 'calendar-todo-item' + (t.status === 'done' ? ' done' : '');
          todoBadge.textContent = t.text; // CSS 규칙 덕분에 글자가 길면 자동으로 ... 처리 됨
          todoBadge.title = t.text; // 마우스 대면 전체 글자 보이게 툴팁 세팅
          cell.appendChild(todoBadge);
        });

        grid.appendChild(cell);
      }

      // 3. 다음달 날짜로 빈칸 채우기 (정사각형 모양 맞추기)
      const totalCells = grid.children.length - 7; // 요일 헤더 제외 수
      const nextMonthCellsNeeded = 42 - totalCells; // 6주 치 칸 채우기 기준
      if(nextMonthCellsNeeded > 0 && nextMonthCellsNeeded < 7) {
        for (let i = 1; i <= nextMonthCellsNeeded; i++) {
          const cell = document.createElement('div');
          cell.className = 'calendar-day-cell other-month';
          cell.innerHTML = `<span class="day-number">${i}</span>`;
          grid.appendChild(cell);
        }
      }
    }

    // 홈 화면 대시보드 업데이트
    function updateHomeSummary() {
      // 2026년 오늘 날짜 포맷 생성
      const todayStr = new Date().toISOString().split('T')[0];
      const todayTasks = tasks.filter(t => t.date === todayStr);
      const remaining = todayTasks.filter(t => t.status !== 'done').length;

      const summaryText = document.getElementById('homeSummaryText');
      if (todayTasks.length === 0) {
        summaryText.innerHTML = `오늘 예정된 마감 일정이 없습니다. 가벼운 하루 되세요! 😊`;
      } else {
        summaryText.innerHTML = `오늘 마감인 일은 총 <strong>${todayTasks.length}건</strong>이며, 그 중 진행중인 일은 <strong>${remaining}건</strong> 남았습니다!`;
      }
    }

    // 데이터 영구 저장 기능 (브라우저를 꺼도 안전함)
    function saveTasks() {
      localStorage.setItem('my_todo_tasks', JSON.stringify(tasks));
    }

    function loadTasks() {
      const stored = localStorage.getItem('my_todo_tasks');
      if (stored) {
        tasks = JSON.parse(stored);
      } else {
        tasks = [];
      }
    }

    // 프로필 정보 설정창 저장
    function saveProfile() {
      profileData.name  = document.getElementById('profileNameInput').value.trim();
      profileData.quote = document.getElementById('profileQuoteInput').value.trim();
      
      localStorage.setItem('my_todo_profile', JSON.stringify(profileData));
      applyProfile();
      showToast('✅ 프로필이 안전하게 영구 저장되었어요!');
    }

    function handleProfilePhoto(event) {
      const file = event.target.files[0];
      if (!file) return;
      const reader = new FileReader();
      reader.onload = function(e) {
        profileData.photo = e.target.result;
        localStorage.setItem('my_todo_profile', JSON.stringify(profileData));
        applyProfile();
        showToast('✅ 프로필 이미지가 등록되었습니다!');
      };
      reader.readAsDataURL(file);
    }

    function applyProfile() {
      const name  = profileData.name  || '이름을 설정하세요';
      const quote = profileData.quote || '뭐든지 할 수 있어!';
      const photo = profileData.photo;

      document.getElementById('profileName').textContent  = name;
      document.getElementById('profileQuote').textContent = quote;
      
      document.getElementById('profileNameInput').value = profileData.name;
      document.getElementById('profileQuoteInput').value = profileData.quote;

      const photoEls = [document.getElementById('profilePhoto'), document.getElementById('profilePhotoPreview')];
      photoEls.forEach(el => {
        if (!el) return;
        el.innerHTML = photo ? '<img src="' + photo + '" alt="프로필">' : '👤';
      });

      // 홈 화면 환영 인사도 커스텀 변경
      if (profileData.name) {
        document.getElementById('welcomeMessage').textContent = `반가워요, ${profileData.name}님! 오늘도 파이팅이에요!🔥`;
      }
    }

    // 디자인 설정값 저장
    function saveDesign() {
      const primary = document.getElementById('colorPrimary').value;
      const sidebar = document.getElementById('colorSidebar').value;
      const font    = document.getElementById('fontFamily').value;

      const designObj = { primary, sidebar, font };
      localStorage.setItem('my_todo_design', JSON.stringify(designObj));
      showToast('🎨 나만의 디자인 색상이 저장되었습니다!');
    }

    function applyDesign() {
      const stored = localStorage.getItem('my_todo_design');
      if (stored) {
        const { primary, sidebar, font } = JSON.parse(stored);
        document.documentElement.style.setProperty('--primary', primary);
        document.documentElement.style.setProperty('--primary-light', primary + '22');
        document.documentElement.style.setProperty('--bg-sidebar', sidebar);
        document.documentElement.style.setProperty('--font-main', font);

        // 디자인 입력창 값들 동기화
        document.getElementById('colorPrimary').value = primary;
        document.getElementById('colorPrimaryValue').textContent = primary;
        document.getElementById('colorSidebar').value = sidebar;
        document.getElementById('fontFamily').value = font;
      }
    }

    function resetDesign() {
      document.documentElement.style.removeProperty('--primary');
      document.documentElement.style.removeProperty('--primary-light');
      document.documentElement.style.removeProperty('--bg-sidebar');
      document.documentElement.style.removeProperty('--font-main');
      
      document.getElementById('colorPrimary').value = '#4f8ef7';
      document.getElementById('colorPrimaryValue').textContent = '#4f8ef7';
      document.getElementById('colorSidebar').value = '#ffffff';
      document.getElementById('fontFamily').value   = "'Noto Sans KR', sans-serif";
      
      localStorage.removeItem('my_todo_design');
      showToast('↩️ 원래 기본 디자인 컬러로 돌아왔습니다.');
    }

    // 데이터 완전 포맷 초기화 기능
    function clearAllData() {
      openModal(
        '데이터 초기화 경고',
        '등록된 할일 리스트, 프로필, 디자인 세팅을 포함한 모든 정보가 지워집니다. 정말 초기화할까요?',
        function() {
          localStorage.clear();
          showToast('🗑 모든 데이터가 파기되었습니다. 초기화 중...');
          setTimeout(() => location.reload(), 800);
        }
      );
    }

    // =====================================================
    // 앱 구동 시작 시점 실행
    // =====================================================
    window.addEventListener('load', function() {
      // 1. 입력창 기본 오늘 날짜 자동 등록 세팅
      const todayString = new Date().toISOString().split('T')[0];
      document.getElementById('todoDateInput').value = todayString;

      // 2. 브라우저 저장소 데이터 불러오기
      loadTasks();

      const storedProfile = localStorage.getItem('my_todo_profile');
      if (storedProfile) profileData = JSON.parse(storedProfile);
      
      // 3. 불러온 데이터 화면 반영 및 세팅
      applyProfile();
      applyDesign();
      updateHomeSummary();
    });
